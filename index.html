async function loadStandings() { try { const response = await fetch("https://statsapi.mlb.com/api/v1/standings?leagueId=103"); const data = await response.json(); // ✅ Use division ID instead of name const alEast = data.records.find(r => r.division.id === 201); if (!alEast) { document.getElementById("standings").innerHTML = "AL East not found"; return; } let html = "

AL EAST STANDINGS

"; let firstPlaceTeam = ""; alEast.teamRecords.forEach((team, index) => { const name = team.team.name; const wins = team.leagueRecord.wins; const losses = team.leagueRecord.losses; if (index === 0) firstPlaceTeam = name; let teamClass = ""; if (name.includes("Red Sox")) teamClass = "sox"; html += `
${index + 1}. ${name} (${wins}-${losses})
`; }); document.getElementById("standings").innerHTML = html; // Banner logic if (firstPlaceTeam.includes("Red Sox")) { banner.innerHTML = "RED SOX ARE IN FIRST PLACE"; banner.className = "banner soxBanner"; } else if (firstPlaceTeam.includes("Yankees")) { banner.innerHTML = "THIS IS A YANKEES FAN WEBSITE NOW"; banner.className = "banner yankeesBanner"; } } catch (e) { console.error(e); document.getElementById("standings").innerHTML = "AL East standings unavailable"; } }
